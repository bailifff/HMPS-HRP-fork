
on_character_creation = {
	on_actions = {
		balance_generals_on_action
	}
}
on_yearly_pulse_country = {
	on_actions = {
    	PDXRP_small_nations_buff 
    }
}
on_game_started = {
	on_actions = {
		PDXRP_all_buff
	}
}
balance_generals_on_action = {
	effect = {
		if = {
			limit = {
				has_role = general
				AND = {
					NOT = {has_trait = basic_offensive_planner}
					NOT = {has_trait = basic_defensive_strategist}
					NOT = {has_trait = basic_artillery_commander}
				}
			}
			random_list = {
				40 = {add_trait = basic_offensive_planner}
				50 = {add_trait = basic_defensive_strategist}
				10 = {add_trait = basic_artillery_commander}
			}
		}
	   else_if = {
			limit = {
				has_role = admiral
				NOT = {has_trait = basic_naval_commander}
			}
			add_trait = basic_naval_commander		

		}
	}
}
PDXRP_small_nations_buff= {
	effect = {
		remove_modifier = modifier_PDXRP_small_nations_buff
		if = {
			limit = {
				global_country_ranking > 10
				NOT = {C:CHI ?= this}
				NOT = {C:BIC ?= this}
			}
			add_modifier = {
				name = modifier_PDXRP_small_nations_buff
				multiplier = sv_small_nations_buff
			}
		}
	}
}
PDXRP_all_buff = {
	effect = {
		every_country = {
			add_modifier = modifier_PDXRP_all_nations
		}
	}
}
